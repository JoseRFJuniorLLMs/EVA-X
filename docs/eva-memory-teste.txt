```go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	"eva-mind/internal/memory" // ajuste o import
)

func main() {
	ctx := context.Background()

	// 1. Total de memórias
	total, err := memory.CountAll(ctx)
	if err != nil {
		log.Fatal(err)
	}
	fmt.Printf("Total memórias: %d\n", total)

	// 2. Redundância aproximada (top 1000 mais recentes)
	duplicates := memory.FindDuplicates(ctx, 0.98, 1000)
	fmt.Printf("Redundantes (>0.98 sim): ~%d%%\n", len(duplicates)*100/1000)

	// 3. Distribuição temporal
	buckets := memory.AgeDistribution(ctx)
	for age, count := range buckets {
		fmt.Printf("%s: %d memórias\n", age, count)
	}

	// 4. Recall@10 (exemplo manual, depois automatize)
	queries := []string{"solidão", "família", "medo", "sono"}
	for _, q := range queries {
```

```go
// internal/memory/types.go
type MemoryMetadata struct {
	AccessCount   int       `json:"access_count"`
	LastAccessedAt time.Time `json:"last_accessed_at"`
	Importance    float64   `json:"importance"` // 0.0–1.0, calculado por analyzer

// internal/memory/retrieval.go
func RetrieveWithBias(ctx context.Context, query string, k int) ([]*Memory, error) {
	// 1. Busca vetorial (Qdrant)
	candidates, _ := qdrant.Search(ctx, embed(query), 100)

	// 2. Re-ranking composto
	scored := make([]ScoredMemory, 0, len(candidates))
	for _, cand := range candidates {
		score := cand.Score * 0.6 // semântica
		score += recencyBoost(cand.LastAccessedAt) * 0.25
		score += accessBoost(cand.AccessCount) * 0.15
		scored = append(scored, ScoredMemory{cand, score})
	}

	sort.Slice(scored, func(i, j int) bool { return scored[i].Score > scored[j].Score })
	topK := scored[:min(k, len(scored))]

	return convertToMemories(topK), nil
}

func recencyBoost(t time.Time) float64 {
	days := time.Since(t).Hours() / 24
	if days < 1 { return 1.0 }
	if days < 7 { return 0.8 }
	if days < 30 { return 0.5 }
	return 0.2 // cold
}

func accessBoost(count int) float64 {
	if count > 50 { return 1.0 }
	if count > 10 { return 0.8 }
	return 0.4
}
```
