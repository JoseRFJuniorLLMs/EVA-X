// generateRandomEmbedding gera um embedding aleatório normalizado
func generateRandomEmbedding(dimension int) []float64 {
	embedding := make([]float64, dimension)
	sum := 0.0
	
	for i := range embedding {
		embedding[i] = rand.NormFloat64()
		sum += embedding[i] * embedding[i]
	}
	
	// Normaliza para ter norma 1 (como embeddings reais)
	norm := math.Sqrt(sum)
	for i := range embedding {
		embedding[i] /= norm
	}
	
	return embedding
}

// generateMemoryDataset gera um conjunto de memórias sintéticas
func generateMemoryDataset(numMemories, dimension int) [][]float64 {
	dataset := make([][]float64, numMemories)
	for i := range dataset {
		dataset[i] = generateRandomEmbedding(dimension)
	}
	return dataset
}

// BASELINE: RECALCULAR TUDO (O(n³))
// ═══════════════════════════════════════════════════════════════════

// fullRecomputation recalcula o subespaço completo usando SVD
func fullRecomputation(memories [][]float64, k int) (*mat.Dense, time.Duration) {
	startTime := time.Now()
	
	numMemories := len(memories)
	dimension := len(memories[0])
	
	// Constrói matriz de memórias
	data := make([]float64, numMemories*dimension)
	for i, mem := range memories {
		copy(data[i*dimension:(i+1)*dimension], mem)
	}
	A := mat.NewDense(numMemories, dimension, data)
	
	// SVD: A = U·Σ·V^T
	var svd mat.SVD
	ok := svd.Factorize(A, mat.SVDThin)
	if !ok {
		panic("SVD failed")
	}
	
	// Extrai as k primeiras colunas de V (direções principais)
	var v mat.Dense
	svd.VTo(&v)
	
	subspace := mat.NewDense(dimension, k, nil)
	for i := 0; i < dimension; i++ {
		for j := 0; j < k; j++ {
			subspace.Set(i, j, v.At(i, j))
		}
	}
	
	elapsed := time.Since(startTime)
	return subspace, elapsed
}

// RANK-1 UPDATE (O(nk))
// ═══════════════════════════════════════════════════════════════════

// incrementalUpdate atualiza o subespaço incrementalmente
func incrementalUpdate(kmm *KrylovMemoryManager, newMemory []float64) time.Duration {
	startTime := time.Now()
	kmm.UpdateSubspace(newMemory)
	return time.Since(startTime)
}

// calculateSubspaceDistance calcula a distância entre dois subespaços
// Usa a norma de Frobenius de (Q₁Q₁^T - Q₂Q₂^T)
func calculateSubspaceDistance(Q1, Q2 *mat.Dense) float64 {
	n, k1 := Q1.Dims()
	_, k2 := Q2.Dims()
	
	if k1 != k2 {
		panic("Subespaços com dimensões diferentes")
	}
	
	// P₁ = Q₁·Q₁^T
	P1 := mat.NewDense(n, n, nil)
	P1.Mul(Q1, Q1.T())
	
	// P₂ = Q₂·Q₂^T
	P2 := mat.NewDense(n, n, nil)
	P2.Mul(Q2, Q2.T())
	
	// Diferença
	diff := mat.NewDense(n, n, nil)
	diff.Sub(P1, P2)
	
	return mat.Norm(diff, 2) // Frobenius norm
}

// calculateRecall calcula o recall@K em busca semântica
func calculateRecall(
	query []float64,
	memoryFull [][]float64,
	memoryCompressed [][]float64,
	k int,
) float64 {
	// Busca no espaço completo (ground truth)
	topKFull := findTopK(query, memoryFull, k)
	
	// Busca no espaço comprimido
	topKCompressed := findTopK(query, memoryCompressed, k)
	
	// Conta quantos estão presentes em ambos
	hits := 0
	for _, idxFull := range topKFull {
		for _, idxComp := range topKCompressed {
			if idxFull == idxComp {
				hits++
				break
			}
		}
	}
	
	return float64(hits) / float64(k)
}

// findTopK encontra os K vetores mais similares (produto interno)
func findTopK(query []float64, memories [][]float64, k int) []int {
	type similarity struct {
		index int
		score float64
	}
	
	similarities := make([]similarity, len(memories))
	
	for i, mem := range memories {
		score := 0.0
		for j := range query {
			score += query[j] * mem[j]
		}
		similarities[i] = similarity{index: i, score: score}
	}
	
	// Ordena por score decrescente (bubble sort simples para teste)
	for i := 0; i < len(similarities)-1; i++ {
		for j := i + 1; j < len(similarities); j++ {
			if similarities[j].score > similarities[i].score {
				similarities[i], similarities[j] = similarities[j], similarities[i]
			}
		}
	}
	
	// Retorna índices dos top-K
	topK := make([]int, k)
	for i := 0; i < k; i++ {
		topK[i] = similarities[i].index
	}
	return topK
}

// TestPrecisionComparison compara precisão: Rank-1 vs Full
func TestPrecisionComparison(t *testing.T) {
	dimension := 1536
	k := 64
	windowSize := 1000
	numMemories := 1000
	
	fmt.Println("\n=== Teste de Precisão: Rank-1 Update vs SVD Completo ===")
	
	// Gera dataset
	memories := generateMemoryDataset(numMemories, dimension)
	
	// Método 1: SVD Completo (ground truth)
	fmt.Println("\n1. Calculando subespaço via SVD completo...")
	subspaceSVD, timeSVD := fullRecomputation(memories, k)
	fmt.Printf("   Tempo: %v\n", timeSVD)
	
	// Método 2: Rank-1 Update Incremental
	fmt.Println("\n2. Calculando subespaço via Rank-1 Updates...")
	startIncremental := time.Now()
	kmm := NewKrylovMemoryManager(dimension, k, windowSize)
	for _, mem := range memories {
		kmm.UpdateSubspace(mem)
	}
	timeIncremental := time.Since(startIncremental)
	fmt.Printf("   Tempo: %v\n", timeIncremental)
	
	// Compara os subespaços
	distance := calculateSubspaceDistance(subspaceSVD, kmm.Basis)
	fmt.Printf("\n3. Distância entre subespaços: %.6f\n", distance)
	
	// Speedup
	speedup := float64(timeSVD) / float64(timeIncremental)
	fmt.Printf("\n4. Speedup: %.1fx mais rápido\n", speedup)
	
	// Verifica se a distância é aceitável
	if distance > 0.1 {
		t.Errorf("Distância muito alta entre subespaços: %.6f (esperado < 0.1)", distance)
	}
	
	// Verifica se houve speedup
	if speedup < 2.0 {
		t.Errorf("Speedup insuficiente: %.1fx (esperado > 2x)", speedup)
	}
}

// TestRecallPreservation verifica que recall@10 é mantido
func TestRecallPreservation(t *testing.T) {
	dimension := 1536
	k := 64
	windowSize := 1000
	numMemories := 500
	
	fmt.Println("\n=== Teste de Recall em Busca Semântica ===")
	
	// Gera dataset
	memories := generateMemoryDataset(numMemories, dimension)
	
	// Constrói subespaço
	kmm := NewKrylovMemoryManager(dimension, k, windowSize)
	for _, mem := range memories {
		kmm.UpdateSubspace(mem)
	}
	
	// Comprime memórias
	compressedMemories := make([][]float64, numMemories)
	for i, mem := range memories {
		compressed, err := kmm.CompressVector(mem)
		if err != nil {
			t.Fatal(err)
		}
		compressedMemories[i] = compressed
	}
	
	// Testa recall com 10 queries aleatórias
	numQueries := 10
	totalRecall := 0.0
	
	for i := 0; i < numQueries; i++ {
		query := generateRandomEmbedding(dimension)
		compressedQuery, _ := kmm.CompressVector(query)
		
		recall := calculateRecall(query, memories, compressedMemories, 10)
		totalRecall += recall
		fmt.Printf("Query %d: Recall@10 = %.2f%%\n", i+1, recall*100)
	}
	
	avgRecall := totalRecall / float64(numQueries)
	fmt.Printf("\nRecall médio: %.2f%%\n", avgRecall*100)
	
	// Verifica se recall é aceitável
	if avgRecall < 0.90 {
		t.Errorf("Recall muito baixo: %.2f%% (esperado >= 90%%)", avgRecall*100)
	}
}

			error := kmm.checkOrthogonality()
			fmt.Printf("Após %d updates: erro = %.6f\n", i+1, error)

// TestScalability verifica comportamento com muitas memórias
func TestScalability(t *testing.T) {
	dimension := 1536
	k := 64
	windowSize := 100000
	
	fmt.Println("\n=== Teste de Escalabilidade ===")
	
	kmm := NewKrylovMemoryManager(dimension, k, windowSize)
	
	scales := []int{1000, 10000, 100000}
	
	for _, scale := range scales {
		startTime := time.Now()
		
		for i := 0; i < scale; i++ {
			embedding := generateRandomEmbedding(dimension)
			kmm.UpdateSubspace(embedding)
		}
		
		elapsed := time.Since(startTime)
		avgTime := elapsed / time.Duration(scale)
		
		fmt.Printf("Escala %d memórias: tempo médio/update = %v\n", 
			scale, avgTime)
		
		// Tempo médio deve ser < 1ms
		if avgTime > time.Millisecond {
			t.Errorf("Performance degradou em escala %d: %v > 1ms", 
				scale, avgTime)
		}
	}
}

// TestMemoryUsage verifica economia de memória
func TestMemoryUsage(t *testing.T) {
	dimension := 1536
	k := 64
	numMemories := 1000
	
	fmt.Println("\n=== Teste de Uso de Memória ===")
	
	// Memória no formato original
	memoryOriginal := numMemories * dimension * 4 // 4 bytes por float32
	fmt.Printf("Memória original (1536D): %d bytes (%.2f MB)\n", 
		memoryOriginal, float64(memoryOriginal)/(1024*1024))
	
	// Memória no formato comprimido
	memoryCompressed := numMemories * k * 4
	fmt.Printf("Memória comprimida (64D): %d bytes (%.2f MB)\n", 
		memoryCompressed, float64(memoryCompressed)/(1024*1024))
	
	// Economia
	reduction := (1.0 - float64(memoryCompressed)/float64(memoryOriginal)) * 100
	fmt.Printf("Redução de memória: %.1f%%\n", reduction)
	
	// Verifica se economia é significativa
	if reduction < 90.0 {
		t.Errorf("Economia de memória insuficiente: %.1f%% (esperado >= 90%%)", 
			reduction)
	}
}
