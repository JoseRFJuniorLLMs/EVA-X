// KrylovMemoryManager gerencia o subespaço dinâmico da EVA
// Utiliza Sliding Window para manter apenas as memórias mais relevantes
type KrylovMemoryManager struct {
	Basis      *mat.Dense    // Matriz de base ortogonal Q (n x k)
	WindowSize int           // Tamanho da janela de memória ativa
	Dimension  int           // Dimensão original dos embeddings (ex: 1536)
	K          int           // Dimensão do subespaço de Krylov (ex: 64)
	
	// Janela Deslizante (FIFO)
	memoryQueue  [][]float64   // Fila circular de memórias
	queueHead    int           // Índice da cabeça da fila
	queueSize    int           // Tamanho atual da fila
	
	// Métricas e Monitoramento
	totalUpdates  int64        // Total de atualizações realizadas
	lastUpdate    time.Time    // Timestamp da última atualização
	updateTimes   []time.Duration // Histórico de tempos de atualização
	
	// Controle de Qualidade
	orthogonalityError float64 // Erro de ortogonalidade atual
	reconstructionError float64 // Erro de reconstrução
	
	// Thread Safety
	mutex sync.RWMutex
}

// NewKrylovMemoryManager cria um novo gerenciador de memória
func NewKrylovMemoryManager(dimension, k, windowSize int) *KrylovMemoryManager {
	return &KrylovMemoryManager{
		Basis:       mat.NewDense(dimension, k, nil),
		WindowSize:  windowSize,
		Dimension:   dimension,
		K:           k,
		memoryQueue: make([][]float64, windowSize),
		queueHead:   0,
		queueSize:   0,
		updateTimes: make([]time.Duration, 0, 100),
		lastUpdate:  time.Now(),
	}
}

// UpdateSubspace adiciona uma nova memória e refina a base de Krylov
// Implementa Rank-1 Update com Gram-Schmidt Modificado
//
// Complexidade: O(n·k) onde n = dimension, k = subspace size
// Muito mais eficiente que recalcular tudo: O(n³)
func (kmm *KrylovMemoryManager) UpdateSubspace(newVector []float64) error {
	startTime := time.Now()
	
	kmm.mutex.Lock()
	defer kmm.mutex.Unlock()
	
	// Validação
	if len(newVector) != kmm.Dimension {
		return fmt.Errorf("dimensão incorreta: esperado %d, recebido %d", 
			kmm.Dimension, len(newVector))
	}
	
	vNew := mat.NewVecDense(kmm.Dimension, newVector)
	
	// ═══════════════════════════════════════════════════════════════
	// ETAPA 1: Gram-Schmidt Modificado
	// ═══════════════════════════════════════════════════════════════
	// Projeta vNew no complemento ortogonal da base atual
	// v_⊥ = v_new - Σ(i=1 to k) ⟨v_new, q_i⟩ q_i
	
	for i := 0; i < kmm.K; i++ {
		basisVec := mat.Col(nil, i, kmm.Basis)
		if basisVec == nil || len(basisVec) == 0 {
			continue // Coluna ainda não inicializada
		}
		
		qVec := mat.NewVecDense(kmm.Dimension, basisVec)
		
		// Produto interno: ⟨v_new, q_i⟩
		dot := mat.Dot(vNew, qVec)
		
		// Subtrai a projeção: vNew = vNew - dot * q_i
		tmp := mat.NewVecDense(kmm.Dimension, nil)
		tmp.ScaleVec(dot, qVec)
		vNew.SubVec(vNew, tmp)
	}
	
	// ═══════════════════════════════════════════════════════════════
	// ETAPA 2: Normalização do Resíduo
	// ═══════════════════════════════════════════════════════════════
	// q_{k+1} = v_⊥ / ‖v_⊥‖
	
	norm := mat.Norm(vNew, 2)
	
	if norm > 1e-6 {
		// Vetor tem informação nova significativa
		vNew.ScaleVec(1.0/norm, vNew)
		
		// ═══════════════════════════════════════════════════════════
		// ETAPA 3: Sliding Window - Shift and Insert
		// ═══════════════════════════════════════════════════════════
		kmm.shiftAndInsert(vNew)
		
		// Atualiza métricas
		kmm.totalUpdates++
		kmm.orthogonalityError = kmm.checkOrthogonality()
		
	} else {
		// Vetor é colinear com a base existente (informação redundante)
		// Não adiciona nada novo - a EVA já "conhece" esse conceito
		fmt.Printf("[Krylov] Vetor redundante detectado (norm=%.6f). Ignorado.\n", norm)
	}
	
	// Métricas de performance
	elapsed := time.Since(startTime)
	kmm.updateTimes = append(kmm.updateTimes, elapsed)
	kmm.lastUpdate = time.Now()
	
	// Mantém apenas últimas 100 medições
	if len(kmm.updateTimes) > 100 {
		kmm.updateTimes = kmm.updateTimes[1:]
	}
	
	return nil
}

// shiftAndInsert implementa a janela deslizante FIFO
// Remove a memória mais antiga e insere a nova na base ortogonal
func (kmm *KrylovMemoryManager) shiftAndInsert(v *mat.VecDense) {
	// Adiciona à fila circular
	kmm.memoryQueue[kmm.queueHead] = v.RawVector().Data
	kmm.queueHead = (kmm.queueHead + 1) % kmm.WindowSize
	
	if kmm.queueSize < kmm.WindowSize {
		kmm.queueSize++
	}
	
	// Determina qual coluna da base substituir
	// Estratégia: Substitui a coluna mais antiga (FIFO)
	colToReplace := kmm.totalUpdates % int64(kmm.K)
	
	// Insere o novo vetor normalizado na base
	kmm.Basis.SetCol(int(colToReplace), v.RawVector().Data)
	
	fmt.Printf("[Krylov] Subespaço atualizado. Coluna %d substituída. "+
		"Fila: %d/%d memórias.\n", 
		colToReplace, kmm.queueSize, kmm.WindowSize)
}

// CompressVector projeta um vetor de alta dimensão no subespaço de Krylov
// Entrada: embedding de 1536D
// Saída: vetor comprimido de 64D
//
// Fórmula: v_compressed = Q^T · v_original
func (kmm *KrylovMemoryManager) CompressVector(vector []float64) ([]float64, error) {
	kmm.mutex.RLock()
	defer kmm.mutex.RUnlock()
	
	if len(vector) != kmm.Dimension {
		return nil, fmt.Errorf("dimensão incorreta: esperado %d, recebido %d",
			kmm.Dimension, len(vector))
	}
	
	v := mat.NewVecDense(kmm.Dimension, vector)
	compressed := mat.NewVecDense(kmm.K, nil)
	
	// Projeção: compressed = Q^T · v
	compressed.MulVec(kmm.Basis.T(), v)
	
	return compressed.RawVector().Data, nil
}

// ReconstructVector reconstrói aproximadamente um vetor original
// a partir de sua forma comprimida
//
// Fórmula: v_reconstructed ≈ Q · v_compressed
func (kmm *KrylovMemoryManager) ReconstructVector(compressed []float64) ([]float64, error) {
	kmm.mutex.RLock()
	defer kmm.mutex.RUnlock()
	
	if len(compressed) != kmm.K {
		return nil, fmt.Errorf("dimensão comprimida incorreta: esperado %d, recebido %d",
			kmm.K, len(compressed))
	}
	
	c := mat.NewVecDense(kmm.K, compressed)
	reconstructed := mat.NewVecDense(kmm.Dimension, nil)
	
	// Reconstrução: reconstructed = Q · c
	reconstructed.MulVec(kmm.Basis, c)
	
	return reconstructed.RawVector().Data, nil
}

// checkOrthogonality verifica quão ortogonal a base Q realmente é
// Uma base perfeitamente ortogonal tem Q^T · Q = I (identidade)
//
// Retorna: ‖Q^T·Q - I‖_F (norma de Frobenius do erro)
func (kmm *KrylovMemoryManager) checkOrthogonality() float64 {
	// Q^T · Q
	gram := mat.NewDense(kmm.K, kmm.K, nil)
	gram.Mul(kmm.Basis.T(), kmm.Basis)
	
	// Subtrai a identidade
	for i := 0; i < kmm.K; i++ {
		current := gram.At(i, i)
		gram.Set(i, i, current-1.0)
	}
	
	// Calcula norma de Frobenius
	return mat.Norm(gram, 2) // Frobenius norm
}

// Reorthogonalize força a reortogonalização completa da base
// Útil quando orthogonalityError > 0.1 (10% de erro)
//
// Usa QR decomposition para garantir ortogonalidade perfeita
func (kmm *KrylovMemoryManager) Reorthogonalize() {
	kmm.mutex.Lock()
	defer kmm.mutex.Unlock()
	
	startTime := time.Now()
	
	// QR decomposition: Basis = Q · R
	var qr mat.QR
	qr.Factorize(kmm.Basis)
	
	// Extrai Q ortogonal
	var q mat.Dense
	q.QFromQR(&qr)
	
	// Substitui a base
	kmm.Basis = &q
	
	elapsed := time.Since(startTime)
	newError := kmm.checkOrthogonality()
	
	fmt.Printf("[Krylov] Reortogonalização completa. "+
		"Erro antes: %.6f, depois: %.6f. Tempo: %v\n",
		kmm.orthogonalityError, newError, elapsed)
	
	kmm.orthogonalityError = newError
}

// GetStatistics retorna estatísticas detalhadas do sistema
func (kmm *KrylovMemoryManager) GetStatistics() map[string]interface{} {
	kmm.mutex.RLock()
	defer kmm.mutex.RUnlock()
	
	// Calcula tempo médio de atualização
	var avgUpdateTime time.Duration
	if len(kmm.updateTimes) > 0 {
		var total time.Duration
		for _, t := range kmm.updateTimes {
			total += t
		}
		avgUpdateTime = total / time.Duration(len(kmm.updateTimes))
	}
	
	// Calcula taxa de compressão
	compressionRatio := float64(kmm.Dimension) / float64(kmm.K)
	memoryReduction := (1.0 - float64(kmm.K)/float64(kmm.Dimension)) * 100
	
	return map[string]interface{}{
		"dimension":           kmm.Dimension,
		"subspace_size":       kmm.K,
		"window_size":         kmm.WindowSize,
		"queue_fill":          kmm.queueSize,
		"total_updates":       kmm.totalUpdates,
		"last_update":         kmm.lastUpdate.Format(time.RFC3339),
		"avg_update_time_us":  avgUpdateTime.Microseconds(),
		"orthogonality_error": kmm.orthogonalityError,
		"compression_ratio":   compressionRatio,
		"memory_reduction_%":  memoryReduction,
		"status":              kmm.getHealthStatus(),
	}
}

// getHealthStatus determina o status de saúde do sistema
func (kmm *KrylovMemoryManager) getHealthStatus() string {
	if kmm.orthogonalityError > 0.1 {
		return "degraded - reorthogonalization needed"
	}
	if kmm.queueSize == 0 {
		return "initializing"
	}
	if kmm.queueSize < kmm.WindowSize/2 {
		return "warming up"
	}
	return "healthy"
}

// MemoryConsolidation executa consolidação periódica da memória
// Deve ser chamado em intervalos regulares (ex: a cada hora)
func (kmm *KrylovMemoryManager) MemoryConsolidation() {
	kmm.mutex.Lock()
	defer kmm.mutex.Unlock()
	
	startTime := time.Now()
	fmt.Println("[Krylov] Iniciando consolidação de memória...")
	
	// 1. Verifica se precisa de reortogonalização
	if kmm.orthogonalityError > 0.05 { // 5% de erro
		kmm.Reorthogonalize()
	}
	
	// 2. Remove memórias muito antigas da fila (opcional)
	// Implementar estratégia de decay aqui se necessário
	
	// 3. Recalcula estatísticas
	kmm.reconstructionError = kmm.calculateReconstructionError()
	
	elapsed := time.Since(startTime)
	fmt.Printf("[Krylov] Consolidação concluída em %v. "+
		"Erro de reconstrução: %.4f\n", elapsed, kmm.reconstructionError)
}

// calculateReconstructionError mede quão bem o subespaço representa as memórias
// Testa compressão + reconstrução em algumas memórias da fila
func (kmm *KrylovMemoryManager) calculateReconstructionError() float64 {
	if kmm.queueSize == 0 {
		return 0.0
	}
	
	// Testa em até 100 memórias aleatórias
	numSamples := min(100, kmm.queueSize)
	totalError := 0.0
	
	for i := 0; i < numSamples; i++ {
		idx := (kmm.queueHead + i) % kmm.WindowSize
		if kmm.memoryQueue[idx] == nil {
			continue
		}
		
		original := kmm.memoryQueue[idx]
		
		// Comprime
		compressed, _ := kmm.CompressVector(original)
		
		// Reconstrói
		reconstructed, _ := kmm.ReconstructVector(compressed)
		
		// Calcula erro L2
		error := 0.0
		for j := 0; j < len(original); j++ {
			diff := original[j] - reconstructed[j]
			error += diff * diff
		}
		totalError += math.Sqrt(error)
	}
	
	return totalError / float64(numSamples)
}

// ═══════════════════════════════════════════════════════════════════
// EXEMPLO DE USO
// ═══════════════════════════════════════════════════════════════════

func ExampleKrylovMemoryManager() {
	// Configuração para EVA-Mind em produção
	dimension := 1536  // Dimensão dos embeddings OpenAI
	k := 64            // Subespaço de Krylov
	windowSize := 10000 // Últimas 10K memórias

	kmm := NewKrylovMemoryManager(dimension, k, windowSize)

	// Simula stream de memórias entrando
	fmt.Println("=== Simulação de Aprendizado Contínuo ===")
	
	for i := 0; i < 1000; i++ {
		// Gera embedding simulado
		embedding := make([]float64, dimension)
		for j := range embedding {
			embedding[j] = math.Sin(float64(i+j) * 0.01)
		}

		// Atualização incremental O(n·k)
		err := kmm.UpdateSubspace(embedding)
		if err != nil {
			fmt.Printf("Erro: %v\n", err)
		}

		// A cada 100 memórias, mostra estatísticas
		if i%100 == 0 {
			stats := kmm.GetStatistics()
			fmt.Printf("\n[Update %d] Estatísticas:\n", i)
			fmt.Printf("  - Tempo médio de atualização: %d µs\n", 
				stats["avg_update_time_us"])
			fmt.Printf("  - Erro de ortogonalidade: %.6f\n", 
				stats["orthogonality_error"])
			fmt.Printf("  - Status: %s\n", stats["status"])
			fmt.Printf("  - Redução de memória: %.1f%%\n\n", 
				stats["memory_reduction_%"])
		}
	}

	// Teste de compressão
	fmt.Println("\n=== Teste de Compressão ===")
	testEmbedding := make([]float64, dimension)
	for i := range testEmbedding {
		testEmbedding[i] = math.Cos(float64(i) * 0.02)
	}

	compressed, _ := kmm.CompressVector(testEmbedding)
	fmt.Printf("Embedding original: %dD\n", len(testEmbedding))
	fmt.Printf("Embedding comprimido: %dD\n", len(compressed))
	fmt.Printf("Taxa de compressão: %.1fx\n", 
		float64(len(testEmbedding))/float64(len(compressed)))

	// Consolidação periódica
	fmt.Println("\n=== Consolidação de Memória ===")
	kmm.MemoryConsolidation()

	// Estatísticas finais
	finalStats := kmm.GetStatistics()
	fmt.Printf("\n=== Estatísticas Finais ===\n")
	for key, value := range finalStats {
		fmt.Printf("%s: %v\n", key, value)
	}
}
