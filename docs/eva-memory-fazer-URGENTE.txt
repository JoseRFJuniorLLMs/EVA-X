```go
// internal/memory/ingestion/pipeline.go
type IngestionPipeline struct {
    embedder *EmbeddingService
    llm      llm.Provider
    store    *GraphStore
}

func (p *IngestionPipeline) ProcessSession(ctx context.Context, sessionText string) error {
    // 1. Extrair factos atómicos + dual timestamp
    facts, err := p.extractAtomicFacts(ctx, sessionText)
    if err != nil {
        return err
    }
    
    for _, fact := range facts {
        // 2. Gerar embedding do facto (não do chunk)
        emb, _ := p.embedder.GenerateEmbedding(ctx, fact.ResolvedText)
        
        // 3. Check relational versioning
        relation := p.classifyRelation(ctx, fact)
        
        // 4. Salvar com metadados ricos
        err := p.store.SaveAtomicFact(ctx, fact, emb, relation)
        if err != nil {
            return err
        }
    }
    
    return nil
}
```

**Dia 4: classifyRelation (Relational Versioning)**  
Antes de salvar, cheque se é update/extend/derive.

```go
func (p *IngestionPipeline) classifyRelation(ctx context.Context, newFact AtomicFact) string {
    similarFacts := p.store.FindSimilarFacts(ctx, newFact.Embedding, 0.85)
    if len(similarFacts) == 0 {
        return "NEW"
    }
    
    prompt := fmt.Sprintf(`
        Classifique a relação entre:
        Novo facto: %s
        Facto existente: %s
        
        Responda apenas: "UPDATE" (contradiz), "EXTEND" (enriquece), "DERIVE" (infere novo), "NONE"
    `, newFact.ResolvedText, similarFacts[0].Text)
    
    resp, _ := p.llm.GenerateText(ctx, &llm.TextRequest{Prompt: prompt})
    return resp.Text // UPDATE / EXTEND / DERIVE / NONE
}
```
